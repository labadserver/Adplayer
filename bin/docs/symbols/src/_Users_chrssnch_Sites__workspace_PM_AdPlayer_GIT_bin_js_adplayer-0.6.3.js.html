200    * @param {url} scriptSrc The url of the script to load.

209   _this.loadScript = function (objId, objReturnFn, scriptSrc, callback) {
210     var jsIntv;
211     var obj;

213     /** If script is not currently being loaded, attempt to load. **/
214     var init = function() {
215       if (checkList(objId)) {
216         function wait() {
217           if(!checkList(objId)) {
218             clearInterval(waitIntv);
219             setScript();
220           }
221         }
222         var waitIntv = setInterval(wait, 100);
223       } else {
224         _loadList.push(objId);
225         setScript();
226       }
227     }
228 
229     /** Attempts to create script element if object does not exist. **/
230     var setScript = function() {
231       if(!checkObj()) {
232         var js = document.createElement('script');
233         js.setAttribute('type', 'text/javascript');
234         js.setAttribute('src', scriptSrc);
235         var s = document.getElementsByTagName('script')[0];
236         s.parentNode.insertBefore(js, s); 
237         jsIntv = setInterval(setObj, 100);
238       } else {
239         setObj();
240       } 
241     }
242 
243     /** Remove from check list executes callback. **/
244     var setObj = function() {
245       if(checkObj()) {
246         clearInterval(jsIntv);
247         removeFromList(objId);      
248         callback();
249       }
250     };
251 
252     /** Checks if function returns valid object. **/
253     var checkObj = function() { 
254       try { 
255           if(objReturnFn()) {
256           return true;
257         }
258        } catch (e) {

260        }
261     };

286          }
287        }
288      };

454    * @function
455    * @description Waits for DOM element to become available and then determines if iframe
456    *              verification is needed or message can be processed.
457    * @param {dom} dom DOM object this needs to be checked.
458    * @param {object} json JSON object to check or pass through.

461     _this.dom = dom;
462     _this.json = json;

464     function iframeVerify(dom, json) {
465       var obj = new Object();
466       obj.postType = PostMessage.INCOMING;
467       obj.uid = json.uid;
468       obj.fn = 'iframePlayerVerify';      
469       var player = _this.getPlayerByDomSearch(dom);
470       if (player) {
471         if (json.fn == "iframePlayerVerify") {
472           obj.params = true;
473           PostMessage.send(obj, dom.contentWindow);

475         }
476         readyTest(dom, json, player);
477       } else {
478         obj.params = false;

480       }

483       Util.log('Could not verify a parent iframe AdPlayer.');

493    */

495     var par = dom.parentNode;

497       par = par.parentNode;

499       if ((parName == 'body') || (parName == 'html')) { break; }
500     }

503       if(adPlayer) {

505           return(adPlayer);
506       } else {

513   /**
514    * @name PostMessageHandler#readyTest
515    * @function
516    * @description Parses JSON object and executes requests.
517    * @param {dom} iframe Target iframe to communicate with.
518    * @param {object} json JSON object.
519    * @param {adplayer} player Current AdPlayer.
520    */
521   function readyTest(iframe, json, player) {
522     if (player) {
523       var params = json.params.split(',');
524       for (var t=0; t < params.length; t++) {
525         // Checks if it contains a function, which is needs to be properly wrapped and send off
526         switch (json.fn){
527           case 'addEventListener':
528             if (unescape(params[t]).match(PostMessage.FUNCTION)) {
529               var funcN = unescape(params[t]).slice(PostMessage.FUNCTION.length);
530               function funcMe (evt) {
531                 var obj = new Object();
532                 obj.postType = PostMessage.INCOMING;
533                 obj.uid = json.uid;
534                 obj.fn = funcN;
535                 obj.evtType = evt.type();
536                 obj.uidName = json.uidName;
537                 PostMessage.send(obj, iframe.contentWindow);
538               }
539               funcMe.uidName = json.uidName;
540               params[t] = funcMe;
541             }
542             break;
543         }
544       }
545       
546       switch (json.fn){
547         case 'removeEventListener':
548             params.push(json.uidName.toString());
549           break;
550         case 'track':
551           adEvtObj = new AdEvent(params[0]);
552           adEvtObj.target(player);
553           adEvtObj.currentTarget(player);
554           player.track(new AdEvent(params[0]), params[1]);
555           return;
556           break;
557       }
558 
559       player[json.fn].apply(player, params);
560     }
561   }

584       case 'defaultTrackCallBack':
585       break;
586     default:

588       var player = AdPlayerManager.getPlayerByUID(json.uid);  
589       // Need to target the current player and send back
590       if (player) {

592         event.target(player);
593         func(event);
594       }
595       break;
596     }
