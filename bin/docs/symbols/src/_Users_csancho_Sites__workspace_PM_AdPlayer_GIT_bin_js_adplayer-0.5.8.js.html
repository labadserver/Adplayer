164   /**
165    * @name Util#loadScript
166    * @function
167    * @description Loads an external script & executes a callback function when an object, located in the 
168    *              external script & is returned in a <code>testFn</code>, is valid. 
169    * @param {string} objId Id used to identify script. Note: Implemented to avoid use of eval();
170    * @param {function} objReturnFn  Function that returns object to check against. Note: Implemented to avoid use of eval();
171    * @param {url} scriptSrc The url of the script to load.
172    * @param {string} callback The handler to be executed when script load is complete.
173    * @example
174    * Util.loadScript('ExtScriptObj', function(){return ExtScriptObj;}, 'http://the.script.url/extobj.js', 
175    *   function(){
176    *      Util.log('External script is done loading.');
177    *   }
178    * );
179    */  
180   _this.loadScript = function (objId, objReturnFn, scriptSrc, callback) {
181     var jsIntv;
182     var obj;
183 
184     /** If script is not currently being loaded, attempt to load. **/
185     var init = function() {
186       if (checkList(objId)) {
187         function wait() {
188           if(!checkList(objId)) {
189             clearInterval(waitIntv);
190             setScript();
191           }
192         }
193         var waitIntv = setInterval(wait, 100);
194       } else {
195         _loadList.push(objId);
196         setScript();
197       }
198     }
199 
200     /** Attempts to create script element if object does not exist. **/
201     var setScript = function() {
202       if(!checkObj()) {
203         var js = document.createElement('script');
204         js.setAttribute('type', 'text/javascript');
205         js.setAttribute('src', scriptSrc);
206         var s = document.getElementsByTagName('script')[0];
207         s.parentNode.insertBefore(js, s); 
208         jsIntv = setInterval(setObj, 100);
209       } else {
210         setObj();
211       } 
212     }
213 
214     /** Remove from check list executes callback. **/
215     var setObj = function() {
216       if(checkObj()) {
217         clearInterval(jsIntv);
218         removeFromList(objId);      
219         callback();
220       }
221     };
222 
223     /** Checks if function returns valid object. **/
224     var checkObj = function() { 
225       try { 
226           if(objReturnFn()) {
227           return true;
228         }
229        } catch (e) {
230         return false;
231        }
232     };
233 
234     /**
235      * Checks loadList for curreent IDs.
236      * @param {string} id String ID to check.
237      * @return Boolean
238      */
239      var checkList = function(id) {
240        for(var i=0; i < _loadList.length; i++) {
241          if (id == _loadList[i]) {
242            return true;
243          }
244        }
245        return false;
246      };
247      
248      /**
249      * Removes an ID from loadList.
250      * @param {string} id The string ID to remove.
251      */
252      var removeFromList = function(id) {
253        for(var i=0; i < _loadList.length; i++) {
254          if (id == _loadList[i]) {
255            _loadList = _loadList.slice(i, 1);
256            return;
257          }
258        }
259      };    
260     
261     init();
262   }

425    * @function
426    * @description Waits for DOM element to become available and then determines if iframe
427    *              verification is needed or message can be processed.
428    * @param {dom} dom DOM object this needs to be checked.
429    * @param {object} json JSON object to check or pass through.

431   _this.domRefPlayerWait = function (dom, json) {
432     _this.dom = dom;
433     _this.json = json;

435     function iframeVerify(dom, json) {
436       var obj = new Object();
437       obj.postType = PostMessage.INCOMING;
438       obj.uid = json.uid;
439       obj.fn = 'iframePlayerVerify';      
440       var player = _this.getPlayerByDomSearch(dom);
441       if (player) {

446         }

462    * @description Attempts to locate a parent AdPlayer from a DOM reference point.

465   _this.getPlayerByDomSearch = function (dom) {
466     var par = dom.parentNode;

468       par = par.parentNode;

471     }
472     if(par) {

474       if(adPlayer) {
475         Util.log('Found player at '+adPlayer.adDomElement().id);
476           return(adPlayer);
477       } else {

479         return null;

484   /**
485    * @name PostMessageHandler#readyTest
486    * @function
487    * @description Parses JSON object and executes requests.
488    * @param {dom} iframe Target iframe to communicate with.
489    * @param {object} json JSON object.
490    * @param {adplayer} player Current AdPlayer.
491    */
492   function readyTest(iframe, json, player) {
493     if (player) {
494       var params = json.params.split(',');
495       for (var t=0; t < params.length; t++) {
496         // Checks if it contains a function, which is needs to be properly wrapped and send off
497         switch (json.fn){
498           case 'addEventListener':
499             if (unescape(params[t]).match(PostMessage.FUNCTION)) {
500               var funcN = unescape(params[t]).slice(PostMessage.FUNCTION.length);
501               function funcMe (evt) {
502                 var obj = new Object();
503                 obj.postType = PostMessage.INCOMING;
504                 obj.uid = json.uid;
505                 obj.fn = funcN;
506                 obj.evtType = evt.type();
507                 obj.uidName = json.uidName;
508                 PostMessage.send(obj, iframe.contentWindow);
509               }
510               funcMe.uidName = json.uidName;
511               params[t] = funcMe;
512             }
513             break;
514         }
515       }
516       
517       switch (json.fn){
518         case 'removeEventListener':
519             params.push(json.uidName.toString());
520           break;
521         case 'track':
522           adEvtObj = new AdEvent(params[0]);
523           adEvtObj.target(player);
524           adEvtObj.currentTarget(player);
525           player.track(new AdEvent(params[0]), params[1]);
526           return;
527           break;
528       }
529 
530       player[json.fn].apply(player, params);
531     }
532   }

541     switch (json.fn){

543         var factoryPlayer;

548           }
549         }

551           factoryPlayer.setIframePlayerType(json);
552         }
553         return;
554       break;      
555       case 'defaultTrackCallBack':
556       break;
557     default: 
558       var func = (new Function( "return( " + json.fn + " );" ))();
559       var player = AdPlayerManager.getPlayerByUID(json.uid);  
560       // Need to target the current player and send back
561       if (player) {
562         var event = new AdEvent(json.evtType);
563         event.target(player);
564         func(event);
565       }
566       break;
567     }
