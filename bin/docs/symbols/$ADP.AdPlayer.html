addEventListener(adEvent, callback)
Adds a callback function to an $ADP.AdEvent flow. Callback handler function returns an $ADP.AdEvent object instance, which contains the following:
  • data - Object containing any data passed at the time of dispatch.
  • type - The $ADP.AdEvent type passed at the time if dispatch.
  • target - Reference to the parent AdPlayer within the setup hierarchy.
  • currentTarget - Reference to the current AdPlayer from where the dispatch orignated.
function trackEventHandler(adEvent) {
 $ADP.Util.log(adEvent.type() + ' has been dispatched');
 $ADP.Util.log(adEvent.data().message);
}
adPlayer.addEventListener($ADP.AdEvent.TRACK, trackEventHandler);

var data = new Object();
data.message = "Hello World!";
adPlayer.track(new $ADP.AdEvent($ADP.AdEvent.TRACK, data), trackEventHandler);

removeTrackingPixel(adEvent, url)
Removes a matching url or callback associated with an $ADP.AdEvent and registered using addTrackingPixel() method.
  • If a callback and a url are both defined, method will remove the $ADP.AdEvent associated with the callback.
  • If only a callback is defined method will remove the $ADP.AdEvent associated with the callback.
  • If only a url is defined, method will remove all $ADP.AdEvent callbacks containing a matching url property.
adPlayer.addTrackingPixel($ADP.AdEvent.TRACK, 'http://my.pixel.url');
adPlayer.addEventListener($ADP.AdEvent.TRACK, removePixels);

function removePixels(adEvent) {
 // Removes url from TRACK event flow.
 adPlayer.removeTrackingPixel($ADP.AdEvent.TRACK, 'http://my.pixel.url');
 
}
